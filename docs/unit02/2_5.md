let currentAns = ""; // 存储当前题目正确答案的简化字符串

function generateProblem() {
    const feedback = document.getElementById('random-feedback');
    const problemArea = document.getElementById('problem-area');
    const input = document.getElementById('user-ans');
    
    feedback.innerText = "";
    input.value = "";

    // 随机生成系数 a 和指数 n: f(x) = ax^n
    const a = Math.floor(Math.random() * 9) + 2; // 2-10
    const n = Math.floor(Math.random() * 5) + 2; // 2-6
    
    // 计算正确答案
    const coeff = a * n;
    const power = n - 1;
    currentAns = `${coeff}x^${power}`;

    // 使用你验证成功的 arithmatex 结构渲染题目
    problemArea.innerHTML = `求导：<span class="arithmatex">\\( f(x) = ${a}x^{${n}} \\)</span>`;

    // 立即触发 MathJax 渲染新生成的公式
    if (window.MathJax && window.MathJax.typesetPromise) {
        window.MathJax.typesetPromise([problemArea]);
    }
}

function checkRandomAns() {
    const userVal = document.getElementById('user-ans').value.replace(/\s+/g, '').toLowerCase();
    const feedback = document.getElementById('random-feedback');
    
    // 简单的字符串比对（你可以根据需要升级为正则或更复杂的逻辑）
    const formattedAns = currentAns.toLowerCase();
    
    if (userVal === formattedAns) {
        feedback.innerHTML = "✅ 太棒了！正确答案正是 <span class='arithmatex'>\\( " + currentAns + " \\)</span>";
        feedback.style.color = "#2e7d32";
    } else {
        feedback.innerHTML = "❌ 不太对，再算算看？提示：系数相乘，指数减一。";
        feedback.style.color = "#d32f2f";
    }

    if (window.MathJax && window.MathJax.typesetPromise) {
        window.MathJax.typesetPromise([feedback]);
    }
}
